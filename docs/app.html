app.get('/:interface/:method', (req, res) ->
  ifc = interfaces[req.param('interface').toLowerCase()]

  Q.all(ifc.InvokeAndMapArguments(req.param('method'), req.query))
  .then((result) -> 
      Util.reply(req, res, result)
  ).fail((err) -> 
      console.error err
      Util.reply(req, res, err)
